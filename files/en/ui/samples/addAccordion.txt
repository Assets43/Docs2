	class Main extends App
	{
		onStart()
		{
			// Creates a fullscreen main layout.
			this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )

			var accords = [ "Accordion 1", "Accordion 2", "Accordion 3" ]

			// Adds an accordion to the main layout
			this.acc = ui.addAccordion( this.main, accords, "", 0.9 )

			// Get the first accordion layout
			this.accLay1 = this.acc.getLayout( 0 )

			// Adds a text control to the first accordion
			ui.addText( this.accLay1, "This is a text in accordion 1." )

			// Get the second accordion layout
			this.accLay2 = this.acc.getLayout( 1 )

			// Adds a button control to the second accordion
			ui.addButton( this.accLay2, "My Button", "Outlined,Secondary,Large" )

			// Get the third accordion layout by its title
			this.accLay3 = this.acc.getLayout( "Accordion 3" )

			// Adds a TextField control to the third accordion
			this.tfd = ui.addTextField( this.accLay3, "", "Outlined" )
    		this.tfd.label = "Type your name"
		}
	}

<sample Add & removing>
	class Main extends App
	{
		onStart()
		{
			// Creates a fullscreen main layout.
			this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )
			this.main.setChildMargins(0, 0.01, 0, 0.01)

			var accords = [ "Accordion 1", "Accordion 2", "Accordion 3" ]

			// Adds an accordion to the main layout
			this.acc = ui.addAccordion( this.main, accords, "", 0.9 )

			// Get the first accordion layout
			this.accLay1 = this.acc.getLayout( 0 )

			// Adds a text control to the first accordion
			ui.addText( this.accLay1, "This is a text in accordion 1." )

			// Get the second accordion layout
			this.accLay2 = this.acc.getLayout( 1 )

			// Adds a button control to the second accordion
			ui.addButton( this.accLay2, "My Button", "Outlined,Secondary,Large" )

			// Get the third accordion layout by its title
			this.accLay3 = this.acc.getLayout( "Accordion 3" )

			// Adds a TextField control to the third accordion
			this.tfd = ui.addTextField( this.accLay3, "", "Outlined" )
			this.tfd.setLabel( "Type your name" )

			// add a button controls to add or remove accordion item
			this.btn1 = ui.addButton( this.main, "Append Item" )
			this.btn1.setOnTouch( this.btn1_onTouch )

			this.btn2 = ui.addButton( this.main, "Add Item in index 1" )
			this.btn2.setOnTouch( this.btn2_onTouch )

			this.btn3 = ui.addButton( this.main, "Remove Item" )
			this.btn3.setOnTouch(  this.btn3_onTouch )
		}

		btn1_onTouch()
		{
			// add an item at the bottom of the accordion
			this.acc.addItem( "New Bottom Title", "New secondary text" )
		}

		btn2_onTouch()
		{
			// insert item in the second index
			this.acc.addItem( "New Second Title", "New secondary text", 1 )
		}

		btn3_onTouch()
		{
			this.acc.removeItemByIndex( 2 )
		}
	}
</sample>
