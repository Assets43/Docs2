

<sample Progress Implementation>
	class Main extends App
    {
        onStart()
        {
            // Creates a fullscreen layout with objects vertically centered.
            this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )
            this.main.setChildSpacing( "Evenly" )

			// Indeterminate Linear
            ui.addProgress( this.main, 40, "linear", 0.7 )

			// Determinate Linear
            ui.addProgress( this.main, null, "Linear,Secondary,indeterminate", 0.7 )

			// Determinate Circular
            ui.addProgress( this.main, 50, "Circular" )

			// Indeterminate Circular, enlarge and secondary
            this.pro = ui.addProgress( this.main, null, "Circular,Secondary,indeterminate" )
            this.pro.circleSize = 100
        }
    }
</sample>
