<sample Bottom Aligned With Margins>

    {

        {
            // Creates a fullscreen main layout. With objects aligned at the bottom.

<sample Slide Layout>
    class Main extends App
    {
        onStart()
        {
            this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )
            this.main.setChildMargins( 0, 0, 0, 0.03 )

            // Add a Slide layout
            this.laySlide = ui.addLayout( this.main, "Slide", "VCenter", 0.3, 0.2 )
            this.laySlide.setBackColor( "rgba(0,0,0,0.1)" )

            // images urls
            var image = "https://upload.wikimedia.org/wikipedia/commons/thumb/9/90/Hapus_Mango.jpg/220px-Hapus_Mango.jpg"
            var image2 = "https://upload.wikimedia.org/wikipedia/commons/thumb/e/ee/Hornet_Mascot.png/640px-Hornet_Mascot.png"

            // Add images component to slide layout
            this.img1 = ui.addImage( this.laySlide, image, "", 1, 1 )
            this.img2 = ui.addImage( this.laySlide, image2, "", 1, 1 )

            // Add a button to the main layout
            this.btn = ui.addButtonGroup( this.main, "Previous,Next" )
            this.btn.setOnTouch( this.onTouch )
        }

        onTouch( e )
        {
            if( e == "Next" ) this.laySlide.next()
            else this.laySlide.previous()
        }
    }
</sample>
