

<sample Basic>
    class Main extends App
    {
        onStart()
        {
            this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )

            this.btn = ui.addButton( this.main, "Show Popup", "Primary" )
            this.btn.setOnTouch( this.ShowPopup )
        }

        ShowPopup()
        {
            ui.showColorPicker( null, "Alpha", this.onSelect )
        }

        onSelect( clr )
        {
            ui.showPopup( clr )
        }
    }
</sample>
