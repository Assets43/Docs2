<sample Callback>
	class Main extends App
	{
		onStart()
		{
			// Creates a fullscreen layout with objects vertically centered.
			this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )

			var list = [
				"Checkbox item 1",
				"Checkbox item 2",
				"Checkbox item 3"
			]

			// Add a checkbox group to the main layout.
			this.ckb = ui.addCheckboxGroup( this.main, list, "Primary", 0.9 )
			this.ckb.setOnTouch( this.onChange )
		}

		onChange( text, checked, value )
		{
			ui.showPopup( text + " : " + checked )
		}
	}
</sample>
