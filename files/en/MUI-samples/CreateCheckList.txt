<sample Light>
cfg.Light
cfg.MUI

function OnStart()
{
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal, "Light")

    lay = MUI.CreateLayout("Linear", "VCenter,FillXY")
    	
    	lst = [
            {name: "Apple", check: false},
            {name: "Mango", check: true},
            {name: "Orange", check: true},
            {name: "Banana"}
        ]
    	chk = MUI.CreateCheckbox("Choose a fruit", lst)
    	chk.SetOnSubmit(OnSubmit)
        chk.Show()

    app.AddLayout(lay)
}

function OnSubmit(lst)
{
    app.ShowPopup(JSON.stringify(lst))   
}
</sample>
