from native import app
from native import gfx

cfg.Light
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = gfx.MUI.CreateLayout("Linear", "VCenter,FillXY")

        tef = gfx.MUI.CreateTEOutlineIconRight(0.8, "Left", "person", "Type your name")
        lay.AddChild(tef)

    app.AddLayout(lay)

<sample SetOnEnter Callback>
from native import app
from native import gfx

cfg.Light
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = gfx.MUI.CreateLayout("Linear", "VCenter,FillXY")

        tef = gfx.MUI.CreateTEOutlineIconRight(0.8, "Left,Number", "call", "Type your number", True)
        tef.SetOnEnter(OnEnter)
        lay.AddChild(tef)

    app.AddLayout(lay)

def OnEnter():
    app.ShowPopup(this.GetText())
</sample>
