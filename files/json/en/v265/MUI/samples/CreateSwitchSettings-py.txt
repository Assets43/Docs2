from native import app
cfg.Light
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    skb = gfx.MUI.CreateSwitchSettings("Enable Sound", 1)
    skb.SetOnTouch(OnTouch)
    lay.AddChild(skb)

    app.AddLayout(lay)

def OnTouch(text, value):
    app.ShowPopup(text+" : "+value)</sample>

from native import app
cfg.Dark
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    settings = [
        "Enable sound",
        "Enable dark mode",
        "Disable full screen"
    ]

    for setting in settings:
        skb = gfx.MUI.CreateSwitchSettings(setting, 1)
        skb.SetOnTouch(OnTouch)
        lay.AddChild(skb)
        lay.AddChild(gfx.MUI.CreateDivider())

    app.AddLayout(lay)

def OnTouch(text, value):
    app.ShowPopup(text+" : "+value)</sample>
