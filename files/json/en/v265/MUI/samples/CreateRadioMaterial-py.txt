<sample Basic>
from native import app
from native import gfx
from native import ui

cfg.Light
cfg.MUI

def OnStart():
    color = ui.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    list = "Apps, Games, Movies, Books"
    rad = ui.CreateRadioMaterial(list)
    lay.AddChild(rad)

    app.AddLayout(lay)
</sample>

<sample With callback>
from native import app
from native import gfx
from native import ui

cfg.Light
cfg.MUI

def OnStart():
    color = ui.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    list = "Apps, Games, Movies, Books"
    rad = ui.CreateRadioMaterial(list, 1, None, "Vertical")
    rad.SetOnTouch(OnTouch)
    lay.AddChild(rad)

    app.AddLayout(lay)

def OnTouch(name, index):
    app.ShowPopup(name + " : " + str(index))
</sample>
