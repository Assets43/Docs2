<sample Light>
from native import app, cfg, ui

cfg.Light
cfg.MUI

def OnStart():
    color = ui.colors.teal
    app.InitializeUIKit(color.lighten1)

    lay = ui.CreateLayout("Linear", "VCenter,FillXY")
    	
    btn = ui.CreateButtonRaisedO("BUTTON", 0.35)
    lay.AddChild(btn)

    btn = ui.CreateButtonRaisedO("[fa-android] ANDROID", 0.35)
    lay.AddChild(btn)

    btn = ui.CreateButtonRaisedO("CUSTOM COLOR", 0.5, 0.1, ui.colors.blue.blue)
    lay.AddChild(btn)

    app.AddLayout(lay)
</sample>
