def OnStart():
    from native import app, gfx, ui

    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.darken1)

    lay = ui.CreateLayout("Linear", "VCenter,FillXY")

    btn = ui.CreateButtonToggle("BUTTON", 0.35, None, False, OnToggle)
    lay.AddChild(btn)

    app.AddLayout(lay)

def OnToggle(v):
    app.ShowPopup(v)
