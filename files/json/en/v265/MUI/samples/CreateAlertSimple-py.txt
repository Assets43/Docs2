from native import app, cfg, ui

cfg.Light
cfg.MUI

def OnStart():
    color = ui.colors.teal
    app.InitializeUIKit(color.teal)

    text = "This is a sample alert text."

    setTimeout(CloseAlert, 4000)

<sample Dark>
from native import app, cfg, ui

cfg.Dark
cfg.MUI

def OnStart():
    color = ui.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "VCenter,FillXY")

    btnShow = ui.CreateButtonRound("SHOW ALERT")
    btnShow.SetOnTouch(ShowAlert)
    lay.AddChild(btnShow)

    app.AddLayout(lay)

    text = "This is a sample alert text."
    als = ui.CreateAlertSimple(text, "GOT IT", ui.colors.green.darken2)
    als.SetOnTouch(AlertOnTouch)

def ShowAlert():
    als.Show()

def AlertOnTouch(val):
    app.ShowPopup(val)
</sample>
