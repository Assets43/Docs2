def OnStart():
    from native import app, gfx, ui

    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "VCenter,FillXY")

    tef = ui.CreateTEOutlineIconLeft(0.8, "Left", "person", "Type your name")
    lay.AddChild(tef)

    app.AddLayout(lay)

<sample SetOnEnter Callback>
cfg.Light
cfg.MUI

def OnStart():
    from native import app, gfx, ui

    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "VCenter,FillXY")

    tef = ui.CreateTEOutlineIconLeft(0.8, "Left,Number", "call", "Type your number", True)
    tef.SetOnEnter(OnEnter)
    lay.AddChild(tef)

    app.AddLayout(lay)

def OnEnter():
    app.ShowPopup(this.GetText())
</sample>
