<sample Basic>
from native import MUI, app

cfg.Light
cfg.MUI

def OnStart():
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = MUI.CreateLayout("Linear", "FillXY,VCenter")

    skb = MUI.CreateSwitch()
    skb.SetOnTouch(OnTouch)
    lay.AddChild(skb)

    app.AddLayout(lay)

def OnTouch(value):
    app.ShowPopup(value)
</sample>

from native import MUI, app

cfg.Light
cfg.MUI

def OnStart():
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal, "Light")

    lay = MUI.CreateLayout("Linear", "FillXY,VCenter")

    skb = MUI.CreateSwitch(True, MUI.colors.blue.blue)
    skb.SetOnTouch(OnTouch)
    lay.AddChild(skb)

    app.AddLayout(lay)

def OnTouch(value):
    app.ShowPopup(value)
