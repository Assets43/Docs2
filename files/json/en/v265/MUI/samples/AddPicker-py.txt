from native import app, cfg, ui

cfg.Light
cfg.MUI

def OnStart():
    lay = ui.CreateLayout("Linear", "VCenter,FillXY")
        	
    ttime = ui.AddTextEdit(lay, 0.7, "outline", "Enter time", True)
    ttime.SetOnTouch(ShowTimePicker)

    tdate = ui.AddTextEdit(lay, 0.7, "outline", "Enter date", True)
    tdate.SetOnTouch(ShowDatePicker)

    tcolor = ui.AddTextEdit(lay, 0.7, "outline", "Enter color", True)
    tcolor.SetOnTouch(ShowColorPicker)

    app.AddLayout(lay)

    tPkr = ui.AddPicker(ttime, "time")
    dPkr = ui.AddPicker(tdate, "date")
    cPkr = ui.AddPicker(tcolor, "color")

def ShowTimePicker():
    tPkr.Show()

def ShowDatePicker():
    dPkr.Show()

def ShowColorPicker():
    cPkr.Show()
