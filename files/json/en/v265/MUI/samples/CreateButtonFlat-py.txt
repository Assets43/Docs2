<sample Light>
from native import app
from native import gfx

cfg.Light
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.lighten1)

    lay = gfx.MUI.CreateLayout("Linear", "VCenter,FillXY")
        
        btn = gfx.MUI.CreateButtonFlat("BUTTON", 0.35)
        lay.AddChild(btn)

        btn = gfx.MUI.CreateButtonFlat("[fa-android] ANDROID", 0.35)
        lay.AddChild(btn)

        btn = gfx.MUI.CreateButtonFlat("CUSTOM COLOR", 0.5, 0.1, gfx.MUI.colors.blue.blue)
        lay.AddChild(btn)

    app.AddLayout(lay)
</sample>
