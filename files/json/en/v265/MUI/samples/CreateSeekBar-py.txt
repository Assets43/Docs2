from native import app, ui

cfg.Light
cfg.MUI

def OnStart():
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    skb = MUI.CreateSeekBar(25, 100, 0.9)
    skb.SetOnTouch(OnTouch)
    lay.AddChild(skb)

    app.AddLayout(lay)

def OnTouch(value):
    app.ShowPopup(value)
