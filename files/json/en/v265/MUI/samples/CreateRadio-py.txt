<sample With callback>
cfg.Light
cfg.MUI

def OnStart():
    from native import app, gfx, ui

    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    list = "Apple, Mango, Banana"
    rad = ui.CreateRadio(list)
    rad.SetOnTouch(OnTouch)
    lay.AddChild(rad)

    app.AddLayout(lay)

def OnTouch(name, index):
    app.ShowPopup(name + " : " + str(index))
</sample>
