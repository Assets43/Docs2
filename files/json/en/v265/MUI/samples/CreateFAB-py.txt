from native import app, cfg, ui

cfg.Light
cfg.MUI

def OnStart():
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Absolute", "FillXY,VCenter")

    fab = ui.CreateFAB("add")
    fab.SetOnTouch(OnClick)
    lay.AddChild(fab)

    app.AddLayout(lay)

def OnClick():
    app.ShowPopup("You clicked FAB")
