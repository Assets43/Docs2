<sample Dark>
from native import app, cfg, ui

cfg.Dark
cfg.MUI

def OnStart():
    color = ui.colors.blue
    app.InitializeUIKit(color.lighten1)

    lay = ui.CreateLayout("Linear", "VCenter,FillXY")
    app.AddLayout(lay)

    ban = ui.CreateBanner("There is no internet connection at the moment.", "LEARN MORE", "CLOSE")
    ban.SetOnTouch(OnTouch)
    ban.Show()

def OnTouch(v):
    if v:
        app.ShowPopup("Learn more")
    else:
        app.ShowPopup("Banner close")
</sample>
