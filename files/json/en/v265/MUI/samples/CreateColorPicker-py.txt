<sample Light>
from native import app
cfg.Light
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    btn = ui.CreateButtonRaised("PICKER", 0.35)
    btn.SetOnTouch(ShowPicker)
    lay.AddChild(btn)

    app.AddLayout(lay)

    clr = gfx.MUI.CreateColorPicker("Choose Color")
    clr.SetOnSelect(OnSelect)

def ShowPicker():
    clr.Show()

def OnSelect(color):
    app.ShowPopup(color)</sample>
