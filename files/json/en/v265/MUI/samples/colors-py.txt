<sample Change Layout Background>
cfg.Light
cfg.MUI
from native import app
from native import gfx
from native import ui

def OnStart():
    color = ui.colors.teal

    lay = ui.CreateLayout("linear", "FillXY,VCenter")

    btn = ui.CreateButtonRaised("Choose Color", 0.4)
    btn.SetOnTouch(ShowDialog)
    lay.AddChild(btn)

    app.AddLayout(lay)

    choices = [
        {"name": "lighten4"},
        {"name": "lighten3"},
        {"name": "lighten2"},
        {"name": "lighten1"},
        {"name": "teal"},
        {"name": "darken1"},
        {"name": "darken2"},
        {"name": "darken3"},
        {"name": "darken4"}
    ]
    lsd = ui.CreateListDialog("Select Color", choices)
    lsd.SetOnSelect(OnSelect)

def ShowDialog():
    lsd.Show()

def OnSelect(name):
    lay.SetBackColor(color[name])
</sample>
