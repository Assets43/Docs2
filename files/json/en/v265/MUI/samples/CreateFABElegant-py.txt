def OnStart():
    from native import app, gfx, ui

    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = ui.CreateLayout("Absolute", "FillXY,VCenter")

    fab = ui.CreateFABElegant("add")
    fab.SetOnTouch(OnClick)
    lay.AddChild(fab)

    app.AddLayout(lay)

def OnClick():
    app.ShowPopup("You clicked FAB")
