from native import app
cfg.Light
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.lighten1)

    lay = ui.CreateLayout("Linear", "VCenter,FillXY")

    btn = ui.CreateButtonRound("BUTTON", 0.35)
    lay.AddChild(btn)

    btn = ui.CreateButtonRound("[fa-android] ANDROID", 0.35)
    lay.AddChild(btn)

    btn = ui.CreateButtonRound("CUSTOM COLOR", 0.5, 0.1, gfx.MUI.colors.blue.blue)
    lay.AddChild(btn)

    app.AddLayout(lay)</sample>
