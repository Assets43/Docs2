from native import app
from native import gfx

cfg.Light
cfg.MUI

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal, "Light")

    lay = gfx.MUI.CreateLayout("Linear", "VCenter,FillXY")

        text = "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Nulla venenatis nunc nec orci finibus blandit. Nam nec consectetur odio. Sed a erat ultrices lectus volutpat malesuada. Aliquam id dictum metus."

        p1 = gfx.MUI.AddText(lay, text, 0.9, None, "Multiline,Paragraph,Light")

        p2 = gfx.MUI.AddTextParagraph(lay, text, 0.9, None, "Multiline", None, "Light")

        p3 = gfx.MUI.CreateTextParagraph(text, 0.9, None, "Multiline", None, "Light")
        lay.AddChild(p3)

    app.AddLayout(lay)
