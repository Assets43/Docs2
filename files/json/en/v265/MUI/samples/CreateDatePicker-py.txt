<sample Default>
from native import app
from native import gfx
from native import ui

cfg.Light
cfg.MUI

def OnStart():
    colors = ui.colors.teal
    app.InitializeUIKit(colors.teal)

    lay = ui.CreateLayout("Linear", "FillXY,VCenter")

    btn = ui.CreateButtonRaised("SHOW PICKER", 0.5)
    btn.SetOnTouch(ShowDatePicker)
    lay.AddChild(btn)

    app.AddLayout(lay)

    pck = ui.CreateDatePicker()
    pck.SetOnSelect(OnSelect)

def ShowDatePicker():
    pck.Show()

def OnSelect(date, yyyy, mm, dd):
    app.ShowPopup(date)
</sample>
