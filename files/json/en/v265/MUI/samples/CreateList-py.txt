<sample Basic>
cfg.Light
cfg.MUI

def OnStart():
    color = MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = MUI.CreateLayout("Linear", "FillXY,VCenter")

        list = ["Downloads", "Images", "Videos"]
        lst = MUI.CreateList(list, 0.95, 0.4)
        lst.SetOnTouch(OnTouch)
        lay.AddChild(lst)

    app.AddLayout(lay)

def OnTouch(title):
    app.ShowPopup(title)
</sample>
