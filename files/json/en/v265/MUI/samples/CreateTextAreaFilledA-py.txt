<sample SetOnEnter Callback>
cfg.Dark
cfg.MUI
from native import app, gfx, ui

def OnStart():
    color = gfx.MUI.colors.teal
    app.InitializeUIKit(color.teal)

    lay = gfx.MUI.CreateLayout("Linear", "VCenter,FillXY")

    tef = gfx.MUI.CreateTextAreaFilledA(0.8, 0.2, "Type your name", True)
    tef.SetOnEnter(OnEnter)
    lay.AddChild(tef)

    app.AddLayout(lay)

def OnEnter():
    app.ShowPopup(this.GetText())
</sample>
