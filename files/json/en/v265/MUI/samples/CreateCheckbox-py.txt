<sample Dark>
cfg.Dark
cfg.MUI

def OnStart():
    lay = MUI.CreateLayout("Linear", "VCenter,FillXY")
    	
    	lst = "Apple,Mango,Orange,Banana"
    	chk = MUI.CreateCheckbox(lst, 0.7)
    	chk.SetOnTouch(OnCheckBox)
    	lay.AddChild(chk)

    app.AddLayout(lay)

def OnCheckBox(index, value):
	app.ShowPopup(lst[index] + " is "+value)
</sample>
