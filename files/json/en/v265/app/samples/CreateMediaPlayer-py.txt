from native import app

def OnStart():
    player = app.CreateMediaPlayer()
    player.SetOnReady(Play)
    player.SetOnComplete(player_OnComplete)
    player.SetFile("/Sys/Snd/Poing.ogg")

def Play():
    player.Play()

def player_OnComplete():
    app.ShowPopup("OnComplete")
