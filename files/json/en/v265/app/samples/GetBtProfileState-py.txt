from native import app

def OnStart():
    states = ["a2dp", "headset", "gatt", "health"]
    lst = []

    for state in states:
        state_value = app.GetBtProfileState(state)
        lst.append(state + ": " + str(state_value))

    app.Alert('\n'.join(lst))
