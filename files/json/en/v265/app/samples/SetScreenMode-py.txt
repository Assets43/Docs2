from native import app, ui

def OnStart():
    lay = app.CreateLayout("Linear", "Vertical,VCenter,FillXY")

    spn = app.CreateSpinner("Default,Normal,Full,Game")
    spn.SetOnChange(SetScreenMode)
    lay.AddChild(spn)

    app.AddLayout(lay)

def SetScreenMode(mode):
    app.SetScreenMode(mode)
