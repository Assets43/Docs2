def OnStart():
    <b>status = app.IsBluetoothEnabled()</b>
    state = app.IsBluetoothOn()

    app.ShowPopup(
        "Bluetooth is " +
        (status ? "enabled" : "disabled") +
        (status == state ? " and " : " but ") +
        (state ? "on" : "off")
    )
