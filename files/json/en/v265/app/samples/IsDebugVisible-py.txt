<sample Toggle Debug>
def OnStart():
    app.CreateDebug()

	lay = app.CreateLayout( "linear", "VCenter,FillXY" )

	tgl = app.CreateToggle( "WiFi Access Point", -1, 0.1 )
	tgl.SetOnTouch( ShowDebug )
	tgl.SetChecked( true )
	lay.AddChild( tgl )

	app.AddLayout( lay )

def ShowDebug( show ):
	app.ShowDebug( show )
	<b>vis = app.IsDebugVisible()</b>
    app.ShowPopup( vis )
