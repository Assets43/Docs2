import math

	lay = app.CreateLayout( "linear", "VCenter,FillXY" )

	lay.AddChild( edt )
	app.AddLayout( lay )

	app.Animate( SimulateKey, 2 )

def SimulateKey():
    ran = 65 + Math.floor(Math.random() * 26)
    char = String.fromCharCode( ran )
