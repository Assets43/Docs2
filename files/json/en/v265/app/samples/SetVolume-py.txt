from native import app

def OnStart():
    lay = app.CreateLayout("linear", "VCenter,FillXY")

    skb = app.CreateSeekBar(1)
    skb.SetRange(1)
    skb.SetOnChange(SetVolume)
    lay.AddChild(skb)

    app.AddLayout(lay)

def SetVolume(value):
    app.SetVolume("Music", value)
    app.PlaySound("/Sys/Snd/Poing.ogg")
