<sample Example>
from native import app

def OnStart():
    app.SetMenu("Start,Stop,Pause")

    lay = app.CreateLayout("linear", "")

    btn = app.CreateButton("[fa-gear]", -1, -1, "fontawesome")
    btn.SetOnTouch(app.ShowMenu)
    lay.AddChild(btn)

    app.AddLayout(lay)

def OnMenu(item):
    app.ShowPopup(item, "Short")
</sample>
