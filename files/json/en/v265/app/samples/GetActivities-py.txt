<sample Show Activities>
def OnStart():
    app.ShowProgress()

    lay = app.CreateLayout( "linear" )

    list = app.GetActivities()

    lst = app.CreateList( "", 1, 1 )
    lay.AddChild( lst )

    app.AddLayout(lay)

    for i in list:
        a = list[i]
        lst.AddItem(a.label, a.packageName)

    app.HideProgress()
