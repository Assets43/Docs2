<sample Exit dialog>
from native import app

def OnStart():
    app.EnableBackKey(False)
    app.ShowPopup("Press the back key")

def OnBack():
    yesNo = app.CreateYesNoDialog("Exit App?")
    yesNo.SetOnTouch(yesNo_OnTouch)

def yesNo_OnTouch(result):
    if(result == "Yes"):
        app.Exit()
