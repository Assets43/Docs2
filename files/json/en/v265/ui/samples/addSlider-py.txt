<sample Slider steps and marks>
class Main(App):
    def onStart(self):
        self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")
        self.sld = ui.addSlider(self.main, 10, "", 0.7)
        self.sld.step = 10
        self.sld.marks = True
        self.sld.setOnSelect(self.onSelect)
        self.txt = ui.addText(self.main, "Value is 10")

    def onSelect(self, value):
        self.txt.text = "Value is " + str(value)
</sample>

<sample Slider with custom step marks>
class Main(App):
    def onStart(self):
        self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")
        self.sld = ui.addSlider(self.main, 10, "Primary", 0.7)
        self.sld.step = 10
        self.sld.marks = [
            { "label": "First", "value": 10 },
            { "label": "Second", "value": 40 },
            { "label": "Third", "value": 60 }
        ]
        self.sld.setOnSelect(self.onSelect)
        self.txt = ui.addText(self.main, "Value is 10")
        self.txt.margins = 0.02

    def onSelect(self, value):
        self.txt.text = "Value is " + str(value)
</sample>
