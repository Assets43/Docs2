from native import ui

def OnStart
    global popup
    main = ui.addLayout("main", "Linear", "VCenter,FillXY")
    sld = ui.addSlider(main, 10, "", 0.7)
    sld.setOnChange(onChange)
    popup = ui.showPopup(10)

def onChange(value):
    popup.text = str(value)
    popup.show()

<sample Slider steps and marks>
from native import ui

def OnStart
    global txt
    main = ui.addLayout("main", "Linear", "VCenter,FillXY")
    sld = ui.addSlider(main, 10, "", 0.7)
    sld.step = 10
    sld.marks = True
    sld.setOnSelect(onSelect)
    txt = ui.addText(main, "Value is 10")

def onSelect(value):
    txt.text = "Value is " + str(value)
</sample>

<sample Slider with custom step marks>
from native import ui

def OnStart
    global txt
    main = ui.addLayout("main", "Linear", "VCenter,FillXY")
    sld = ui.addSlider(main, 10, "Primary", 0.7)
    sld.step = 10
    sld.marks = [
        { "label": "First", "value": 10 },
        { "label": "Second", "value": 40 },
        { "label": "Third", "value": 60 }
    ]
    sld.setOnSelect(onSelect)
    txt = ui.addText(main, "Value is 10")
    txt.margins = 0.02

def onSelect(value):
    txt.text = "Value is " + str(value)
</sample>
