from native import ui

def onStart(self):
    self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")

    items = ["This is item 1", "This is item 2", "This is item 3"]

    self.swg = ui.addSwitchGroup(self.main, items, "Secondary", 0.9)
    self.swg.setOnTouch(self.onTouch)

def onTouch(self, item, value):
    ui.showPopup(item + " : " + value)

<sample Elevated switchgroup with onChange callback>
from native import ui

def onStart(self):
    global items
    self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")

    items = ["This is item 1", "This is item 2", "This is item 3"]

    self.swg = ui.addSwitchGroup(self.main, items, "Secondary,Elevated", 0.9)
    self.swg.label = "My label"
    self.swg.elevation = 4

    self.swg.setOnChange(self.onChange)

def onChange(self, items, indexes):
    ui.showPopup(", ".join(items))
</sample>
