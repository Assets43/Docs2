                }

<sample Manipulate dom inside the webview>
from native import ui

def onStart(self):
    global loaded, web
    self.loaded = False
    self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")
    self.web = ui.addWebView(self.main, "", "", 0.7, 0.5)
    self.web.setOnLoad(self.onLoad)
    self.web.html = '''
        <html>
        <head></head>
        <body>
            <h2 id="my-text">Hi from webview!</h2>
        </body>
        </html>
    '''
    self.btn = ui.addButton(self.main, "Change background color")
    self.btn.setOnTouch(self.onTouch)

def onLoad(self):
    global loaded
    self.loaded = True

def onTouch(self):
    if self.loaded:
        dom = self.web.document
        dom.body.style.backgroundColor = "green"
        h2 = dom.getElementById("my-text")
        h2.innerText = "New text from parent!"
        h2.style.color = "white"
    else:
        ui.showPopup("Webview is not yet loaded!")
</sample>
