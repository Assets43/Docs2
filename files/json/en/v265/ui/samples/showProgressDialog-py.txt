<sample Nocancel progress dialog>
from native import ui

def onStart(self):
    self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")
    self.btn = ui.addButton(self.main, "Show Progress Dialog", "Outlined")
    self.btn.setOnTouch(self.onTouch)

def onTouch(self):
    self.pdlg = ui.showProgressDialog("Loading...", "NoCancel")

    def hideProgressDialog():
        self.pdlg.hide()
        ui.showPopup("Progress dialog is close!")

    setTimeout(hideProgressDialog, 2000)
</sample>
