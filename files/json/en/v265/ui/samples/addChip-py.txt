from native import app
from native import ui

class Main(app.App):
    def onStart(self):
        # Creates a fullscreen layout with objects vertically centered.
        self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")
        self.main.setChildMargins(0, 0.025, 0, 0.025)

        # Add a default chip to the main layout
        self.chip1 = ui.addChip(self.main, "Basic Chip")

        # Add a callback handler for `onTouch` event.
        self.chip1.setOnTouch(self.onTouch)

        # Add a small primary chip
        self.chip2 = ui.addChip(self.main, "Small Primary Chip", "Primary,Small")
        self.chip2.setOnTouch(self.onTouch)

        # Add a touchable secondary chip
        self.chip3 = ui.addChip(self.main, "Secondary Chip", "Secondary,Touchable")
        self.chip3.setOnTouch(self.onTouch)

    def onTouch(self):
        ui.showPopup("You touched the chip!")

sample = Main()
sample.start()
