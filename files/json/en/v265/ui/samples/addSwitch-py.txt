<sample Basic>
from native import app
from native import ui

class Main(app):
    def onStart(self):
        # Creates a layout with objects verticaly centered
        self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")

        # Adds a switch to the main layout
        self.swt = ui.addSwitch(self.main, "Check me", "Secondary", 0.5)

        # Add a callback handler when the switch control is touch
        self.swt.setOnTouch(self.onTouch)

    def onTouch(self, value):
        ui.showPopup("Value is : "  + value)

        # Setting the switch label text
        self.swt.text = f"Check me (Value is {value})"

main = Main()
main.onStart()

<sample Right switch with custom styles>
from native import app
from native import ui

class Main(app):
    def onStart(self):
        # Creates a layout with objects verticaly centered
        self.main = ui.addLayout("main", "Linear", "VCenter,FillXY")

        # Adds a switch control to the main layout
        # `Right` option will display the switch on the right
        self.swt = ui.addSwitch(self.main, "Check me", "Right,Secondary", 0.5)

        # Add some custom styling to thw switch control
        self.swt.border = 1
        self.swt.padding = ["1.2rem"]
        self.swt.cornerRadius = 4
        self.swt.backColor = "yellow"

        # Add a callback handler when the switch control is touch
        self.swt.setOnTouch(self.onTouch)

    def onTouch(self, value):
        ui.showPopup("Value is : " + value)

main = Main()
main.onStart()
