

<sample Chips>
	class Main extends App
	{
		onStart()
		{
			// Creates a fullscreen layout with objects vertically centered.
			this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )
			this.main.setChildMargins( 0.01, 0.01, 0.01, 0.01 )

			// Basic Chip
			this.chip1 = ui.addChip( this.main, 'Basic Chip' )

			// Primary Small chip
			this.chip2 = ui.addChip( this.main, 'Primary Chip Small', 'Primary,Small' )

			// Touchable chip
			this.chip3 = ui.addChip( this.main, 'Secondary Chip', 'Secondary,Touchable' )

			// Deletable chip
			this.chip4 = ui.addChip( this.main, 'Deletable', 'Secondary,Touchable')
			this.chip4.setOnDelete( this.onDelete )

			// Chip with icon
			this.chip5 = ui.addChip( this.main, 'Shopping Cart' )
			this.chip5.icon = 'shopping_cart'

			// Chip with avatar
			this.chip6 = ui.addChip( this.main, 'Character Avatar', 'Primary,Touchable' )
			this.chip6.avatar = 'E'
		}

		onDelete()
		{
			ui.showPopup( "Deleted" )
		}
	}
</sample>
