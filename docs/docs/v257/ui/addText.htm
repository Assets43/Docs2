Example - Basic Text Control

class Main extends App
    {
        onStart()
        {
            // Create a full screen layout.
            this.main = ui.addLayout( "main", "Linear", "VCenter,FillXY" )

            var t = "This is the sample text to be displayed."

            this.text = ui.addText( this.main, t )
        }
    }

Example - Alignments & Colors

class Main extends App
    {
        onStart()
        {
            // Create a full screen layout.
            this.main = ui.addLayout( "main", "Linear", "VCenter", 1, 1 )

            var t = "This is a text to display."

            // Default is left
            this.text = ui.addText( this.main, t, "body1,Left", 0.9 )

            // Center and color primary
            this.text = ui.addText( this.main, t, "body1,Center,Primary", 0.9 )

            // Right and color secondary
            this.text = ui.addText( this.main, t, "body1,Right,Secondary", 0.9 )
        }
    }
